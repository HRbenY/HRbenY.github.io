const pages = dv.pages('"日志"').sort(p => p.file.day); // "日志" 替换为你存放日记的文件夹名称
const current = pages.find(p => p.file.path === dv.current().file.path);
const index = pages.indexOf(current);

if (index !== -1) {
    const prev = pages[index - 1];
    const next = pages[index + 1];
    
    dv.paragraph(`
    << ${prev ? "[[" + prev.file.name + "|上一篇]]" : "没有了"} | ${current.file.day.toFormat("yyyy-MM-dd")} | ${next ? "[[" + next.file.name + "|下一篇]]" : "未来"} >>
    `);
}
